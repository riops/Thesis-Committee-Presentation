\section{Introduction \& Motivation}

\begin{frame}{Motivation}
    \begin{itemize}
        \item \textbf{Gauge-Gravity Duality:} Connections established between chaotic systems and black hole physics have motivated research into chaotic dynamics in quantum systems.
        \item \textbf{Measuring Chaos:}
        \begin{itemize}
            \item Classical: Lyapunov exponents (exponential divergence of trajectories).
             % 
            \item Quantum: Out-of-Time-Ordered Correlators (OTOCs) where $C(t) \sim e^{2\lambda_L t}$.
        \end{itemize}
        \item \textbf{The MSS Bound:} Maldacena, Shenker, and Stanford suggested a universal upper bound on quantum chaos:
        \begin{equation*}
            \lambda_L \le \frac{2\pi k_B T}{\hbar}
        \end{equation*}
        Black holes are expected to saturate this bound.
    \end{itemize}
\end{frame}

\begin{frame}{Thesis Goals}
    \begin{block}{Primary Goal}
        To study the quantum level dynamics, starting from near thermal equilibrium configurations in Bosonic Matrix (BM) models on Fuzzy 2-Spheres ($S_F^2 \times \mathbb{R}^1$).
        % 
    \end{block}

    \begin{block}{Specific Objectives}
        \begin{enumerate}
            \item Apply \textbf{Gaussian State Approximation (GSA)} to average Heisenberg equations of motion to obtain the equations of motion for the 1-point and 2-point correlators.
            \item Solve the resulting truncated differential equations numerically.
            \item Calculate the \textbf{Largest Lyapunov Exponent ($\lambda_L$)} to determine chaotic behavior.
            \item Investigate the relationship between $\lambda_L$, Energy, and Temperature to check the MSS bound.
        \end{enumerate}
    \end{block}
\end{frame}

% Section 2: The Model
\section{Model and Methods}

\begin{frame}{Hamiltonian on $S_F^2 \times \mathbb{R}$}
    The Hamiltonian for the matrix models considered is expressed as:
    \begin{equation*}
        \hat{H} = \frac{1}{2} Tr \left[ \hat{P}_i^2 + \mu^2 \hat{X}_i^2 - \frac{[L_a, \hat{X}_i]^2}{R^2} - \frac{\lambda}{2} \sum_{i \neq j} [\hat{X}_i, \hat{X}_j]^2 \right]
    \end{equation*}
   
    \begin{itemize}
        \item $\hat{X}_i$: $N \times N$ Hermitian matrices.
        \item $\hat{P}_i$: $N \times N$ Hermitian conjugate momentum operators i.e. 
        \begin{align*}
            [\hat{X}_i, \hat{P}_j] = i\delta_{ij} \mathbb{I}_{N^2\times N^2}.
        \end{align*}
        \item $\mathcal{L}_a := [L_a, \cdot]$: Derivatives on $S_F^2$ satisfying $SU(2)$ algebra.
        \item $L_a$: Spin $\frac{N-1}{2}$ IRR of $SU(2)$.
    \end{itemize}
    
    The system exhibits $SU(2)$ symmetry and, when planar/non-planar terms are included, $SO(2)$ symmetry and $D_4$ when non-planar terms are excluded.
\end{frame}

\begin{frame}{Real Polarization Operator Basis}
    To facilitate numerical solution, the Hamiltonian is expanded using the real polarization operator basis $Y_{lm}$:
    \begin{equation*}
        \hat{X}_i = \hat{X}_i^{lm} Y_{lm}, \quad \hat{P}_i = \hat{P}_i^{lm} Y_{lm}.
    \end{equation*}
    Which is given in terms of the Polarization Operator Basis $T_{lm}$ as:
    \begin{equation*}
    Y_{lm} = 
    \begin{cases}
      \frac{1}{\sqrt{2}} (T_{lm} + T_{lm}^\dagger) & \text{if } m > 0 \\
        T_{l0} & \text{if } m = 0 \\
        \frac{i}{\sqrt{2}} (T_{lm} - T_{lm}^\dagger) & \text{if } m < 0
    \end{cases}, 
    \end{equation*}
    with the derivatives on $S_F^2$ given by:
    \begin{equation*}
      \mathcal{L}_3  Y_{lm} = [L_3, Y_{lm}] = m Y_{lm}
    \end{equation*}
    \begin{equation*}
      \mathcal{L}^2  Y_{lm} = [L_a, [L_a, Y_{lm}]] = -[L_a, Y_{lm}]^2 = l(l+1) Y_{lm}
    \end{equation*}
   
\end{frame}

\begin{frame}
  \frametitle{Heisenberg Equations of Motion}
  We can compute the Heisenberg equations of motion using
  \begin{equation*}
    \dot{\xi}_{i}^{l_1 m_1} = i [H, \xi_{i}^{l_1 m_1}] \text{, with } [X_i^{l_1 m_1}, P_j^{l_2 m_2}] = i\delta_{ij} \delta^{l_1 l_2} \delta^{m_1 m_2}
  \end{equation*}
  This gives us the following equations of motion:
    \begin{align*}
        \dot{X}_{1}^{lm} &= P_{1}^{l m}, \;\dot{X}_{2}^{lm} = P_{2}^{l m},  \\
        \dot{P}_1^{lm} &= -\left[\left(\mu^2 + \frac{l(l+1)}{R^2}\right) X_1^{l m} + \frac{\lambda}{2} \mathcal{N}^{l\; l_1 l_2 l_3}_{m\; m_1 m_2 m_3} X_1^{l_1 m_1} X_2^{l_2 m_2} X_2^{l_3 m_3}\right] \\
        \dot{P}_2^{lm} &= -\left[\left(\mu^2 + \frac{l(l+1)}{R^2}\right) X_1^{l m} + \frac{\lambda}{2} \tikzmarknode{N}{\mathcal{N}}^{l\; l_1 l_2 l_3}_{m\; m_1 m_2 m_3} X_2^{l_1 m_1} X_1^{l_2 m_2} X_1^{l_3 m_3}\right].
    \end{align*}

\begin{frame}{Gaussian State Approximation (GSA)}
    To inspect the real time dynamics of the system, we use the Gaussian State Approximation (GSA) to average the Heisenberg equations of motion, using the following Wigner function of the Gaussian density matrix:
    \begin{equation*}
        W(\xi) = \mathcal{N} e^{-\frac{1}{2}(\xi - \braket{\xi})^T \Sigma^{-1} (\xi - \braket{\xi})}
    \end{equation*}
    where the covariance matrix is given by:
    \begin{equation*}
        \Sigma = \begin{pmatrix}
            \braket{\braket{X_i^{lm} X_j^{l'm'}}} & \braket{\braket{X_i^{lm} P_j^{l'm'}}} \\
            \braket{\braket{X_i^{lm} P_j^{l'm'}}} & \braket{\braket{P_i^{lm} P_j^{l'm'}}}
        \end{pmatrix}
    \end{equation*}
    \begin{itemize}
        \item \textbf{Wick's Theorem:} Used to express the equations of motion via 1-point and 2-point correlation functions.
        \item \textbf{Result:} A coupled system of truncated differential equations for $\langle X \rangle$, $\langle P \rangle$, and correlators like $\langle XX \rangle$, $\langle XP \rangle$ and $\langle PP \rangle$.
    \end{itemize}
\end{frame}

\begin{frame}{Gaussian Averaged EOM}
    The equations of motion for the correlators are derived via Gaussian averaging. For example:
    \begin{align*}
        \partial_t \langle X_1^{lm} \rangle &= \langle P_1^{lm} \rangle \\
        \partial_t \langle P_1^{lm} \rangle &= -\left[ \left(\mu^2 + \frac{l(l+1)}{R^2}\right) \langle X_1^{lm} \rangle + \frac{\lambda}{2} \mathcal{N} \dots \right]
    \end{align*}
    
    The dynamics of the 2-point correlators (e.g., $\partial_t \langle \langle X X \rangle \rangle$) are similarly derived, recovering classical equations if 2-point correlators are zero.
\end{frame}

\begin{frame}
  \frametitle{Thermal States}
    To construct the thermal states, we need to use the symmetries of the system to find the appropriate initial conditions.
    \begin{itemize}
        \item \textbf{Symmetry:} Parity and Time Reversal symmetry.
        \item \textbf{Result:}
        \begin{align*}
            \langle X_1^{lm} \rangle = \langle X_2^{lm} \rangle = 0,\;\langle P_1^{lm} \rangle = \langle P_2^{lm} \rangle = 0,\;\braket{X_i^{lm} P_j^{l'm'}} = 0
        \end{align*}
        \item \textbf{Symmetry:} $SO(2)$ or $D_4$ symmetry.
        \item \textbf{Result:}
        \begin{align*}
            \braket{X_i^{lm} X_j^{lm}} \sim \delta_{ij}
        \end{align*}
        \item \textbf{Symmetry:} $SU(2)$ symmetry.
        \item \textbf{Result:}
        \begin{align*}
            \braket{X_i^{lm} X_j^{l'm'}} = \sigma_{xx}(l) \delta_{ij} \delta^{ll'} \delta^{mm'},\; \braket{P_i^{lm} P_j^{l'm'}} = \sigma_{pp}(l) \delta_{ij} \delta^{ll'} \delta^{mm'}
        \end{align*}
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Entropy}
    \begin{itemize}
        \item von Neumann entropy for Gaussian density matrix can be written in terms of the symplectic eigenvalues.
        \item Symplectic eigenvalues can be written using the eigenvalues of the $\Sigma \Omega$ matrix. ($\Omega$: Symplectic form of canonical coordinates.)
        \item Eigenvalues of the matrix come in complex pairs $\lambda_{[k]} = \pm i f_{[k]}$, where the $f_{[k]}$ are the symplectic eigenvalues.
    \end{itemize}
    \begin{align*}
        S &= \sum_{[k]} \left(f_{[k]} + \frac{1}{2}\right)\ln\left(f_{[k]} + \frac{1}{2}\right) - \sum_{[k]} \left(f_{[k]} - \frac{1}{2}\right)\ln\left(f_{[k]} - \frac{1}{2}\right), \\
          &= \sum_{l} (2l+1)\left[ \left(f_{l} + \frac{1}{2}\right)\ln\left(f_{l} + \frac{1}{2}\right) - \left(f_{l} - \frac{1}{2}\right)\ln\left(f_{l} - \frac{1}{2}\right) \right]
    \end{align*}
\end{frame}

\begin{frame}
    \frametitle{Symplectic Eigenvalues}
    \begin{itemize}
        \item Using these initial conditions we can write the entropy from the symplectic matrix
    \end{itemize}
    \fontsize{7pt}{11pt}\selectfont
    \begin{align*}
        (\Sigma \Omega)_{i j}^{[l m][l' m']} &= \Sigma_{i k}^{[l m][l'' m'']} \Omega_{k j}^{[l'' m''][l' m']} \nonumber \\
        &= \begin{pmatrix}
            \sigma_{xx}(l) \delta^{l l'} \delta^{m m'} \delta_{ik} & 0 \\[6pt]
            0 & \sigma_{pp}(l) \delta^{l l'} \delta^{m m'} \delta_{ik}
        \end{pmatrix} 
        \begin{pmatrix}
            0 & \delta^{l'' l'} \delta^{m'' m'} \delta_{kj} \\[6pt]
            -\delta^{l'' l'} \delta^{m'' m'} \delta_{kj} & 0
        \end{pmatrix} \nonumber \\
        &= \begin{pmatrix}
            0 & \sigma_{xx}(l) \delta^{l l'} \delta^{m m'} \delta_{ij} \\[6pt]
            -\sigma_{pp}(l) \delta^{l l'} \delta^{m m'} \delta_{ij} & 0
        \end{pmatrix}.
    \end{align*}
    \fontsize{9pt}{11pt}\selectfont
    \begin{itemize}
        \item The symplectic eigenvalues are then 
    \end{itemize}
    \begin{align*}
        f_{[k]} = \pm \sqrt{\sigma_{xx}(l) \sigma_{pp}(l)}.
    \end{align*}
    \begin{itemize}
        \item ${[k]} \equiv [i,l,m] = 1,2,\ldots,2N^2$. With only $N$ independent symplectic eigenvalues.
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Energy}
    The energy of the system is given by $E=\braket{H}$. If we use the initial conditions given previously, we get:
    \begin{align*}
        E = \sum_{l=0}^{N-1} (2l+1) \left( \sigma_{pp}(l) + \left(\mu^2 + \frac{l(l+1)}{R^2}\right) \sigma_{xx}(l) \right) + \frac{\lambda C^2}{2N}
    \end{align*}
    Here $C$ is 
    \begin{equation*}
        C := \sum_{l=0}^{N-1} (2l+1) \sigma_{xx}(l) =  \frac{\braket{\braket{Tr\left(X_i X_i\right)}}}{2}
    \end{equation*}
    

\end{frame}

\begin{frame}
    \frametitle{Entropy Maximization}
    We need to maximize the Entropy $S$ for a fixed energy $E$. We can use the method of Lagrange multipliers to find the maximum of 
    \begin{equation*}
        \mathcal{L} = S + \beta \left(E - E_0\right)
    \end{equation*}
    We can maximize the entropy by taking the derivative of $\mathcal{L}$ with respect to $\sigma_{xx}(l)$ and $\sigma_{pp}(l)$ and setting it to zero, we get:
    \begin{equation*}
        \frac{\sigma_{pp}(l)}{\sigma_{xx}(l)} = \frac{\omega_l}{N^2}
    \end{equation*}
    where $\omega_l := N \sqrt{\mu^2 + \frac{l(l+1)}{R^2} + \frac{\lambda C}{N}}$. We also have the symplectic eigenvalue equation:
    \begin{equation*}
        \sigma_{xx}(l) \sigma_{pp}(l) = f_l^2
    \end{equation*}
    which can be solved to give
    \begin{equation*}
        f_l = \frac{1}{2} coth \left(\frac{\beta \omega_l}{N}\right)
    \end{equation*}
\end{frame}

\begin{frame}{Consistency Equation}
    Now we know that the symplectic eigenvalues for the maximum entropy state, all we have to do is to find the value of $C$ for a given $\beta$ and system parameters.
    \begin{block}{Consistency Equation}
        The relationship between the $C$ value and temperature $\beta$ is determined by solving:
        \begin{equation*}
             C = \frac{1}{2} \sum_{l=0}^{N-1} (2l+1) \frac{\coth(\beta \omega_l / N)}{\omega_l}
        \end{equation*}
        where $\omega_l$ depends on $\mu$, $R$, and $C$ itself.
    \end{block}
    Solving this equation allows us to exactly determine $f_l$ for a given $\beta$ and system parameters.
\end{frame}

\begin{frame}{Selecting Initial Conditions}
    \begin{block}{Static Solution}
        By plugging in the values $\sigma_{xx}(l) = \frac{N f_l}{\omega_l}$ and $\sigma_{pp}(l) = \frac{\omega_l f_l}{N}$ in our equations of motion, we can check the Thermal equilibrium is also the static solution of the system. 
    \end{block}
    Therefore, we need a way to choose initial conditions while also ensuring the system is dynamical. To do so, we can separate the variance into quantum and classical dispersions:
    \begin{equation*}
        \sigma_{\xi \xi}(l) = \sigma_{\xi \xi}^0(l) + \sigma_{\xi \xi}^c(l)
    \end{equation*}
   
    \begin{itemize}
        \item \textbf{Quantum Dispersion ($\sigma^0$):} Zero-temperature limit.
        \item \textbf{Classical Dispersion ($\sigma^c$):} Difference between thermal and zero-temp limits.
    \end{itemize}
    One-point initial conditions are selected randomly from a Gaussian distribution with variance $\sigma^c$ to ensure the system is dynamical.
\end{frame}

\section{Computational Methods \& Results}

\begin{frame}{Computational Methods}
    \begin{itemize}
        \item \textbf{Implementation:} C++ code structure involving \texttt{Analysis}, \texttt{Systems}, and \texttt{LyapunovAnalysis} modules.
        \item \textbf{Integration Scheme:} Symplectic Euler Method.
        \begin{itemize}
            \item Preserves the symplectic structure of the system.
            \item Update order: $P \to XP \to PP \to XX \to X$.
        \end{itemize}
        \item \textbf{Lyapunov Calculation:} Renormalize-and-sum procedure.
        \begin{equation*}
            \lambda_L \approx \frac{1}{n \Delta t} \sum_{i=1}^n \ln \left( \frac{|\delta \xi(t_i)|}{|\delta \xi(0)|} \right)
        \end{equation*}
       % 
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Computational Methods: Integration Scheme}
    The integration scheme is given by the following equations:
    \begin{align*}
        \braket{P}_{n+1}  &= \braket{P}_n \;\;\,+ h F^{(2)}\left(\braket{X}_n, \braket{\braket{XX}}_n\right)\\
        \braket{XP}_{n+1} &= \braket{XP}_n + h F^{(4)}\left(\braket{X}_n, \braket{\braket{XX}}_n, \braket{\braket{PP}}_n\right)\\
        \braket{PP}_{n+1} &= \braket{PP}_n + h F^{(5)}\left(\braket{X}_n, \braket{\braket{XX}}_n, \braket{\braket{XP}}_{n+1}\right)\\
        \braket{XX}_{n+1} &= \braket{XX}_n + h F^{(3)}\left(\braket{\braket{XP}}_{n+1}\right)\\
        \braket{X}_{n+1}  &= \braket{X}_n \;\;\, + h F^{(1)}\left(\braket{P}_{n+1}\right)
    \end{align*}
    This leads to the better conservation of the symplectic structure of the system compared to the Runge-Kutta method.

\end{frame}
